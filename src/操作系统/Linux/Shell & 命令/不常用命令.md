```shell
# 刷新文件系统缓冲区
sync

# 重启（安全）
reboot
	-f 强制关机，不调用shutdown
	-p 切断电源

# 关机或重启（安全）
shutdown
  -r 重启
  now 立即执行，默认1分钟后关机
  +10 10分钟后执行
  06:00 凌晨6点执行
	-c 取消关机

# 初始化系统（不安全）
init
  0 关机
  3 字符界面模式
  5 图形界面模式
  6 重启
# 查看当前运行模式，会显示两个数字，第一个数字表示上一次的模式，第二个数字表示当前的模式，上一次的模式为空时表示为N
runlevel

# 停止所有进程并关闭系统（不安全）
halt
# 关闭电源（不安全）
poweroff
```

```shell
# 修改提示符的格式（命令行前面那一部分），可以用来区分生产环境和测试环境，在一定程度上避免误操作
# 超级管理员的命令提示符默认是#，普通用户的命令提示符默认是$

# 查看
echo $PS1

# 临时修改
PS1 = '...'

# 永久修改，在配置文件中添加命令，配置文件中的命令会在用户登录时执行
# 方式一：custom_prompt.sh这个文件是我们自己建的，随时可以删除
echo 'PS1="..."' > /etc/profile.d/custom_prompt.sh
# 重新执行文件，重新登录也可以
source /etc/profile.d/custom_prompt.sh
# 方式二：.bashrc时系统自带的文件，尽量不要修改
echo 'PS1="..."' >> ~/.bashrc
# 重新执行文件，重新登录也可以
source ~/.bashrc
```

```shell
# 输出1到100，分隔符默认为换行符
seq 1 100
# 指定分隔符为+号
seq -s+ 1 100
```

```shell
# 转换和删除字符（注意，是字符，不是字符串），命令执行时才会要求输入原始字符串
tr <字符串> [字符串]
	-d 删除所有属于第一字符集的字符
	-s 把连续重复的字符以单独一个字符表示，即去重
	-t 将第一个字符集对应字符转化为第二字符集对应的字符
	-c 取字符集的补集

# 将1替换为a，将2替换为b，将3替换为c
tr 123 abc

# 删除所有非abc的字符
tr -dc abc

# 倒序字符串
rev
echo 123 | rev
```

```shell
# 将标准输入，输出到多个地方，例如：文件、标准输出、管道...
tee

# 例子：控制台会打印123，且文件也会有123
echo 123 | tee <file>
# 例子：后面的命令会收到123，且文件也会有123
echo 123 | tee <file> | <command>
```
